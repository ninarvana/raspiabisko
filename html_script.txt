    .gallery { display: flex; flex-wrap: wrap; justify-content: center; gap: 15px; margin-top: 20px; }
    .item { width: 200px; }
    .item img { width: 100%; border-radius: 8px; box-shadow: 0 0 4px rgba(0,0,0,0.2); }

<body>
  <h1>Galerie photo Raspberry</h1>
  <div class="gallery" id="gallery"></div>
  <script>
    fetch(".")
      .then(response => response.text())
      .then(text => {
        const parser = new DOMParser();
        const html = parser.parseFromString(text, "text/html");
        const links = Array.from(html.querySelectorAll("a"))
          .map(a => a.getAttribute("href"))
          .filter(href => {
  const lower = href.toLowerCase();
  return (lower.endsWith(".jpg") || lower.endsWith(".jpeg") || lower.endsWith(".png")) && !lower.includes("index.html");
});

        const gallery = document.getElementById("gallery");
        links.forEach(file => {
          const item = document.createElement("div");
          item.className = "item";

          const img = document.createElement("img");
          img.src = file;
          img.alt = file;

          const name = document.createElement("div");
          name.className = "filename";
          name.textContent = file;

          item.appendChild(img);
          item.appendChild(name);
          gallery.appendChild(item);
        });
      });
  </script>
</body>
